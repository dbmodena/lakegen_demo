import pandas as pd

# Load the datasets using the exact paths provided
happiness_df_path = '/home/gabriele.martinelli/LakeGen/LakeGen_T/Data/data_csv/2019.csv'
cost_living_df_path = '/home/gabriele.martinelli/LakeGen/LakeGen_T/Data/data_csv/Cost_of_Living_Index_by_Country_2024.csv'

df_happiness = pd.read_csv(happiness_df_path)
df_cost = pd.read_csv(cost_living_df_path)

# Identify the top 5 countries by happiness score from the 2019 dataset
top_5_happiness = df_happiness.sort_values(by='Score', ascending=False).head(5)

# Merge the top 5 happiness data with the cost of living index data
# We use an inner join to ensure we only return countries that exist in both datasets,
# which also avoids presenting NaN values for missing cost of living indices.
merged_df = pd.merge(
    top_5_happiness, 
    df_cost, 
    left_on='Country or region', 
    right_on='Country', 
    how='inner'
)

# Select only the required columns for the final answer
result = merged_df[['Country or region', 'Cost of Living Index']]

# Check if the resulting dataframe is empty and print the appropriate output
if result.empty:
    print("ERROR_EMPTY: No matching records found for those filters")
else:
    print(result)

import pandas as pd

# Load the datasets using the exact paths provided
path_2019 = '/home/gabriele.martinelli/LakeGen/LakeGen_T/Data/data_csv/2019.csv'
path_cost = '/home/gabriele.martinelli/LakeGen/LakeGen_T/Data/data_csv/Cost_of_Living_Index_by_Country_2024.csv'

df_2019 = pd.read_csv(path_2019)
df_cost = pd.read_csv(path_cost)

# Normalize the country columns to lowercase and strip whitespace to ensure a successful join
df_2019['Country or region'] = df_2019['Country or region'].str.lower().str.strip()
df_cost['Country'] = df_cost['Country'].str.lower().str.strip()

# Merge the two dataframes on the country columns
merged_df = pd.merge(df_2019, df_cost, left_on='Country or region', right_on='Country')

# Filter for countries with a happiness score (Score) above 6 
# and a Local Purchasing Power Index (LPI) below 50
filtered_df = merged_df[(merged_df['Score'] > 6) & (merged_df['Local Purchasing Power Index'] < 50)]

# Calculate the number of countries that meet the criteria
result = len(filtered_df)

# Output the final answer or an error message if no records match
if result == 0:
    print("ERROR_EMPTY: No matching records found for those filters")
else:
    print(result)
