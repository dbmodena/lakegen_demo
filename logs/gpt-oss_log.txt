import pandas as pd

# Load datasets
happiness = pd.read_csv('/home/gabriele.martinelli/LakeGen/LakeGen_T/Data/data_csv/2019.csv')
cost_of_living = pd.read_csv('/home/gabriele.martinelli/LakeGen/LakeGen_T/Data/data_csv/Cost_of_Living_Index_by_Country_2024.csv')

# Merge on country names (handle potential naming mismatches)
merged = pd.merge(
    happiness,
    cost_of_living,
    left_on='Country or region',
    right_on='Country',
    how='inner'
)

# Get top 5 countries by happiness score
top5 = merged.nlargest(5, 'Score')[['Country or region', 'Score', 'Cost of Living Index']]

# Output the result
print(top5)
