def solve():

    # 1. Process 2019 European Elections Data
    try:
        df_2019 = pd.read_csv(path_2019, sep=';')
    except Exception:
        # Fallback for comma separator if semicolon fails
        df_2019 = pd.read_csv(path_2019)
    
    df_2019.columns = df_2019.columns.str.lower().str.strip()

    # We group by 'zona' and sum 'totale_votan'
    df_2019_summary = df_2019.groupby('zona')['totale_votan'].sum().reset_index()
    df_2019_summary.rename(columns={'totale_votan': 'totale_elettori_2019'}, inplace=True)
    df_2019_summary['zona'] = df_2019_summary['zona'].astype(str).str.lower().str.strip()

    # 2. Process 2020 Regional Elections Data
    try:
        df_2020 = pd.read_csv(path_2020_candidates, sep=';')
    except Exception:
        df_2020 = pd.read_csv(path_2020_candidates)
        
    df_2020.columns = df_2020.columns.str.lower().str.strip()

    for cand in candidates:

import pandas as pd

def solve():

    # 1. DATA INGESTION
    try:
        df_agri = pd.read_csv(agri_path)
    except Exception as e:
        print(f"Error loading agri table: {e}")
        return

    try:
        df_pop = pd.read_csv(pop_path)
    except Exception as e:
        print(f"Error loading pop table: {e}")
        return

    # 2. DEFENSIVE CLEANING
    df_agri.columns = df_agri.columns.str.lower().str.strip()
    df_pop.columns = df_pop.columns.str.lower().str.strip()

    # Check required columns
    required_agri = ['latitudine', 'area_statistica']
    required_pop = ['area_statistica', 'residenti', 'eta']
    
    for col in required_agri:
        if col not in df_agri.columns:
            print(f"Error: Column {col} not found in agri table. Available: {df_agri.columns.tolist()}")
            return
            
    for col in required_pop:
        if col not in df_pop.columns:
            print(f"Error: Column {col} not found in pop table. Available: {df_pop.columns.tolist()}")
            return

    # 3. PROCESSING POPULATION DATA
    # We need to identify minors. We'll extract the starting age from the 'eta' string.
    # 'eta' might be '0-1', '1-2', '18-19', etc.
    df_pop['age_start'] = df_pop['eta'].str.extract(r'(\d+)').astype(float)

    df_pop['is_minor'] = df_pop['age_start'] < 18

    # Aggregate population by area_statistica
    # We need total residents and total minor residents per area
    pop_agg = df_pop.groupby('area_statistica').agg(
        total_residents=('residenti', 'sum'),
        minor_residents=('residenti', lambda x: x[df_pop.loc[x.index, 'is_minor']].sum())
    ).reset_index()

    # Calculate percentage of minors
    pop_agg['perc_minori'] = (pop_agg['minor_residents'] / pop_agg['total_residents']) * 100

    # 4. JOINING DATA
    # Cast join keys to string to prevent mismatches
    df_agri['area_statistica'] = df_agri['area_statistica'].astype(str).str.strip()

    # Merge agri with population stats
    df_merged = pd.merge(
        df_agri, 
        pop_agg[['area_statistica', 'perc_minori']], 
        on='area_statistica', 
        how='inner'
    )

    if df_merged.empty:

    try:
        # Load datasets
        df_farms = pd.read_csv(path_farms)
        df_pop = pd.read_csv(path_pop)
    except FileNotFoundError as e:
        print(f"Error: File not found. {e}")
        return

    # Standardize column names to lowercase
    df_farms.columns = df_farms.columns.str.lower()
    df_pop.columns = df_pop.columns.str.lower()

    # Defensive cleaning: ensure join keys are strings and stripped
    df_farms['area_statistica'] = df_farms['area_statistica'].astype(str).str.strip()
    df_pop['area_statistica'] = df_pop['area_statistica'].astype(str).str.strip()

    # Process population data to identify minors and calculate percentages
    # Extract numeric part of 'eta' to identify age < 18
    df_pop['age_num'] = pd.to_pop_numeric = pd.to_numeric(
        df_pop['eta'].str.extract('(\d+)', expand=False), errors='coerce'
    )
    
    # Ensure residenti is numeric
    df_pop['residenti'] = pd.to_numeric(df_pop['residenti'], errors='coerce').fillna(0)

    # Calculate total residents and minor residents per area
    pop_summary = df_pop.groupby('area_statistica').agg(
        total_res=('residenti', 'sum')
    ).reset_index()

    minor_pop = df_pop[df_pop['age_num'] < 18].groupby('area_statistica')['residenti'].sum().reset_index()
    minor_pop.columns = ['area_statistica', 'minor_res']

    # Merge totals and minors
    pop_summary = pop_summary.merge(minor_pop, on='area_statistica', how='left').fillna(0)
    
    # Calculate percentage of minors
    pop_summary['pct_minors'] = (pop_summary['minor_res'] / pop_summary['total_res']) * 100

    # Process farm data: get average latitude per area
    df_farms['latitudine'] = pd.to_numeric(df_farms['latitudine'], errors='coerce')
    farm_summary = df_farms.groupby('area_statistica')['latitudine'].mean().reset_index()

    # Merge farm latitude with population percentage
    merged = pd.merge(farm_summary, pop_summary, on='area_statistica', how='inner')

    if merged.empty:
        print("Error: No overlapping data found between farms and population areas.")
        return

    # Calculate the Pearson correlation coefficient
    correlation = merged['latitudine'].corr(merged['pct_minors'])

    # Output result
    if pd.isna(correlation):

    else:
        print(f"The correlation between the latitude of agricultural businesses and the percentage of minors in the same area is: {correlation}")
