def solve():

    # 1. Process 2019 European Elections Data
    try:
        df_2019 = pd.read_csv(path_2019, sep=';')
    except Exception:
        # Fallback for comma separator if semicolon fails
        df_2019 = pd.read_csv(path_2019)
    
    df_2019.columns = df_2019.columns.str.lower().str.strip()

    # We group by 'zona' and sum 'totale_votan'
    df_2019_summary = df_2019.groupby('zona')['totale_votan'].sum().reset_index()
    df_2019_summary.rename(columns={'totale_votan': 'totale_elettori_2019'}, inplace=True)
    df_2019_summary['zona'] = df_2019_summary['zona'].astype(str).str.lower().str.strip()

    # 2. Process 2020 Regional Elections Data
    try:
        df_2020 = pd.read_csv(path_2020_candidates, sep=';')
    except Exception:
        df_2020 = pd.read_csv(path_2020_candidates)
        
    df_2020.columns = df_2020.columns.str.lower().str.strip()

    for cand in candidates:
