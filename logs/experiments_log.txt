import pandas as pd

def solve():
    # File paths
    path_zones = '/home/gabriele.martinelli/LakeGen/lakegen_demo/Data/bologna_update/datasets/csv/zone-30kmh-bologna.csv'
    path_traffic = '/home/gabriele.martinelli/LakeGen/lakegen_demo/Data/bologna_update/datasets/csv/rilevazione-flusso-veicoli-tramite-spire-anno-2024.csv'

    # Load datasets
    try:
        df_zones = pd.read_csv(path_zones)
        df_traffic = pd.read_csv(path_traffic)
    except FileNotFoundError as e:
        return

    # Identify hourly columns (pattern XX_XX_XX_XX)
    # These columns represent the vehicle count for each hour of the day
    hourly_cols = [col for col in df_traffic.columns if col.count('_') == 3 and col.split('_')[0].isdigit()]

    # Calculate the total volume for each record (sum of all hourly counts)
    df_traffic['daily_volume'] = df_traffic[hourly_cols].sum(axis=1)

    # Aggregate data by sensor (codice_spira) to get the total volume for each sensor in 2024
    # We also keep 'nome_via' to facilitate the join with the zones table
    df_sensor_totals = df_traffic.groupby(['codice_spira', 'nome_via'])['daily_volume'].sum().reset_index()

    # Prepare zone names for matching (normalized)
    # We use the 'nome' column from the zone-30kmh table
    zone_names = df_zones['nome'].str.lower().str.strip().unique()

    # Prepare sensor street names for matching (normalized)
    # We use the 'nome_via' column from the traffic table
    df_sensor_totals['nome_via_clean'] = df_sensor_totals['nome_via'].str.lower().str.strip()

    # Filter sensors that are located within the 'Zone 30kmh'
    # A sensor is considered in the zone if its street name matches a name in the zone table
    mask = df_sensor_totals['nome_via_clean'].isin(zone_names)
    df_sensors_in_zone = df_sensor_totals[mask]

    # Calculate the average total volume for these sensors
    if not df_sensors_in_zone.empty:
        average_volume = df_sensors_in_zone['daily_volume'].mean()
    else:
        average_volume = 0.0

    print(average_volume)

if __name__ == "__main__":

    # Load datasets
    try:
        df_traffic = pd.read_csv(traffic_file)
        df_varchi = pd.read_csv(varchi_file)

    except FileNotFoundError as e:
        print(f"Error: File not found - {e}")
        return

    # 1. Identify the hourly traffic columns (e.g., '00_00_01_00')
    # These columns represent the traffic volume in specific time slots.
    # We identify them by checking if the column name consists of digits and underscores.
    hourly_cols = [col for col in df_traffic.columns if col.replace('_', '').isdigit() and len(col) >= 8]

    # 2. Identify sensors (varchi) that are within 'Zone 30kmh'
    # Since we cannot perform a spatial join with geo_shape using only Pandas,

    zone_names = df_zones['nome'].dropna().unique().tolist()
    
    # Create a regex pattern to match any of the zone names
    if not zone_names:

    import re
    pattern = '|'.join([re.escape(name) for name in zone_names])

    # Filter varchi that belong to the 30km/h zones based on text description
