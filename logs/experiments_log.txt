def solve():

import pandas as pd

def solve():
    # File paths
    evp_path = '/home/gabriele.martinelli/LakeGen/lakegen_demo/data/paris/datasets/csv/plub_evp.csv'
    arbre_path = '/home/gabriele.martinelli/LakeGen/lakegen_demo/data/paris/datasets/csv/plub_arbre.csv'

    # Load necessary columns from plub_evp.csv
    # n_sq_evp: unique identifier for the green space
    # st_area_shape: area of the green space
    evp_cols = ['n_sq_evp', 'st_area_shape']
    df_evp = pd.read_csv(evp_evp_path := evp_path, usecols=evp_cols)

    # Load necessary columns from plub_arbre.csv
    # n_sq_ap: identifier for the green space where the tree is located
    # num_plub: identifier for the remarkable tree
    arbre_cols = ['n_sq_ap', 'num_plub']
    df_arbre = pd.read_csv(arbre_path, usecols=arbre_cols)

    # Ensure ID columns are strings for a robust join
    df_evp['n_sq_evp'] = df_evp['n_sq_evp'].astype(str)
    df_arbre['n_sq_ap'] = df_arbre['n_sq_ap'].astype(str)

    # Join the tables to find trees located within protected green spaces
    # We use the specific asymmetric keys identified in the schema
    merged_df = pd.merge(
        df_evp, 
        df_arbre, 
        left_on='n_sq_evp', 
        right_on='n_sq_ap', 
        how='inner'
    )

    # 1. Total number of remarkable trees in these spaces
    total_trees = len(merged_df)

    # 2. Average area of the protected green spaces that contain these trees
    # We must calculate the mean of the unique green spaces, not the mean of the trees' habitats
    # (to avoid weighting the average by the number of trees in each space)
    avg_area_spaces = merged_df.drop_duplicates('n_sq_evp')['st_area_shape'].mean()

    # Prepare the final answer
    result_str = (
        f"La superficie moyenne des espaces verts protégés contenant des arbres remarquables est de "
        f"{avg_area_spaces:.2f} m², et il y a un total de {total_trees} arbres remarquables."
    )

    print(result_str)

if __name__ == "__main__":
    solve()
