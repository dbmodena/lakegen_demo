def solve():

    # We need n_sq_ess for the join and n_sq_ca for potential grouping/context
    try:
        ess_df = pd.read_csv(ess_path, usecols=['n_sq_ess', 'n_sq_ca'])

    except ValueError:
        evp_df = pd.read_csv(evp_path)

    # Data Cleaning: Cast ID columns to string to avoid type mismatch during merge

import pandas as pd
import re

def solve():
    # 1. DATA INGESTION
    # Using the exact file paths provided in the prompt
    plub_path = '/home/gabriele.martinelli/LakeGen/lakegen_demo/data/paris/datasets/csv/plub_arbre.csv'
    arbres_path = '/home/gabriele.martinelli/LakeGen/lakegen_demo/data/paris/datasets/csv/arbresremarquablesparis.csv'

    try:
        # We only need the activity (a08_nom_us) and the zone/sector (a08_nom_sc)
        df_plub = pd.read_csv(plub_path, usecols=['a08_nom_us', 'a08_nom_sc'])
        # We need the arrondissement (com_arrondissement) to link the trees to the zones
        df_arbres = pd.read_csv(arbres_path, usecols=['com_arrondissement'])
    except Exception:
        # Fallback in case of path variations in different environments
        df_plub = pd.read_csv(plub_path, usecols=['a08_nom_us', 'a08_nom_sc'])
        df_arbres = pd.read_csv(arbres_path, usecols=['com_arrondissement'])

    # 2. DATA PROCESSING
    # To link the sectors (zones) in plub_arbre to the trees in arbresremarquablesparis,
    # we extract the arrondissement number from both datasets.
    # Since we don't have a spatial join, the 'zone' is linked via its arrondissement.
    
    def extract_arrondissement_id(text):
        if pd.isna(text):
            return None
        # Find the first sequence of digits in the string (e.g., "1er" -> "1")
        match = re.search(r'(\d+)', str(text))
        return match.group(1) if match else None

    # Create a common key for joining: the arrondissement number as a string
    df_plub['arr_id'] = df_plub['a08_nom_sc'].apply(extract_arrondissement_id)
    df_arbres['arr_id'] = df_arbres['com_arrondissement'].apply(extract_arrondissement_id)

    # Ensure IDs are strings to avoid type mismatch
    df_plub['arr_id'] = df_plub['arr_id'].astype(str)
    df_arbres['arr_id'] = df_arbres['arr_id'].astype(str)

    # 3. ROBUST JOIN
    # Join the protected activities/zones with the trees based on the arrondissement ID
    # We use an inner join to only keep activities that have corresponding tree data in the same arrondissement
    merged_df = pd.merge(df_plub, df_arbres, on='arr_id', how='inner')

    # 4. AGGREGATION
    # Group by the activity and the specific zone (sector) and count the number of trees
    # The count represents the number of remarkable trees found in the arrondissement of that zone.
    result = merged_df.groupby(['a08_nom_us', 'a08_nom_sc']).size().reset_index(name='nb_arbres_total')

    # 5. OUTPUT FORMATTING
    if result.empty:
        print("Aucune donnée correspondante trouvée pour les activités et zones protégées.")
    else:
        # Prepare a clean, readable string
        output_list = []
        for _, row in result.iterrows():
            output_list.append(f"Activité: {row['a08_nom_us']} | Zone: {row['a08_nom_sc']} | Arbres: {row['nb_arbres_total']}")
        
        # Handle large datasets by summarizing
        if len(output_list) > 10:
            print(f"Nombre total d'activités trouvées: {len(output_list)}")
            print("Exemples des premières lignes :")
            print("\n".join(output_list[:5]))
        else:
            print("\n".join(output_list))
