result = None

if not tv_shows.empty:
    # The 'cast' column contains comma-separated actors.
    # We split the strings into lists and explode them so each actor is a separate row.
    # We use vectorized string methods to ensure robustness against NaN values.
    tv_shows['cast'] = tv_shows['cast'].str.split(',')
    
    # Explode the list of actors into individual rows
    exploded_cast = tv_shows.explode('cast')
    
    # Clean the actor names by stripping leading/trailing whitespace
    exploded_cast['cast'] = exploded_cast['cast'].str.strip()
    
    # Remove rows where cast is NaN or an empty string (which can occur from trailing commas)
    exploded_cast = exploded_cast.dropna(subset=['cast'])
    exploded_cast = exploded_cast[exploded_cast['cast'] != '']
    
    if not exploded_cast.empty:
        # Count the frequency of each actor
        actor_counts = exploded_cast['cast'].value_counts()
        
        if not actor_counts.empty:
            # Identify the actor with the highest frequency

else:

import pandas as pd

# Define the exact paths provided in the instructions
netflix_path = '/home/gabriele.martinelli/LakeGen/lakegen_demo/Data/data_csv/netflix_titles.csv'
cost_path = '/home/gabriele.martinelli/LakeGen/lakegen_demo/Data/data_csv/Cost_of_Living_Index_by_Country_2024.csv'

def calculate_average_rent_index():
    # Load the Netflix titles dataset
    df_netflix = pd.read_csv(netflix_path)

    # The 'country' column can contain multiple countries separated by ', '.
    # To accurately count titles per country, we split and explode the column.
    df_netflix['country'] = df_netflix['country'].str.split(', ')
    df_netflix = df_netflix.explode('country')

    # Count the total number of titles (movies and shows) produced by each country
    country_counts = df_netflix.groupby('country').size().reset_index(name='title_count')

    # Filter for countries that produced more than 500 total titles
    countries_over_500 = country_counts[country_counts['title_count'] > 500]

    # Load the Cost of Living Index dataset
    df_cost = pd.read_csv(cost_path)

    # Join the filtered countries with the Cost of Living data
    # Netflix column is 'country', Cost of Living column is 'Country'
    merged_df = pd.merge(
        countries_over_500, 
        df_cost, 
        left_on='country', 
        right_on='Country'
    )

    # Calculate the average 'Rent Index' for these countries
    if not merged_df.empty:
        average_rent_index = merged_df['Rent Index'].mean()
        print(average_rent_index)
    else:
        print("No countries found matching the criteria.")

if __name__ == "__main__":
    calculate_average_rent_index()
