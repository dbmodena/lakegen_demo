import pandas as pd

def solve():
    # File paths
    ess_path = '/home/gabriele.martinelli/LakeGen/lakegen_demo/data/paris/datasets/csv/plub_ess.csv'
    evp_path = '/html/gabriele.martinelli/LakeGen/lakegen_demo/data/paris/datasets/csv/plub_evp.csv'
    # Note: The prompt provided a path for evp that looks like a typo in the prompt's text vs the logic, 
    # but I will use the path provided in the AVAILABLE TABLES section.
    evp_path = '/home/gabriele.martinelli/LakeGen/lakegen_demo/data/paris/datasets/csv/plub_evp.csv'

    try:
        # Load ESS data
        df_ess = pd.read_csv(ess_path)
        # Load EVP data
        df_evp = pd.read_csv(evp_path)
    except Exception as e:
        print(f"Error loading files: {e}")
        return

    # Standardize column names to lowercase
    df_ess.columns = df_ess.columns.str.lower()
    df_evp.columns = df_evp.columns.str.lower()

    # Check for required columns
    required_ess = ['n_sq_ess', 'n_sq_ca']
    required_evp = ['n_sq_ca']
    
    for col in required_ess:
        if col not in df_ess.columns:
            print(f"Error: Column {col} not found in ESS table. Available: {df_ess.columns.tolist()}")
            return
            
    for col in required_evp:
        if col not in df_evp.columns:
            print(f"Error: Column {col} not found in EVP table. Available: {df_evp.columns.tolist()}")
            return

    # Cast join key to string to prevent mismatch
    df_ess['n_sq_ca'] = df_ess['n_sq_ca'].astype(str)
    df_evp['n_sq_ca'] = df_evp['n_sq_ca'].astype(str)

    # Perform inner join on n_sq_ca
    # This identifies ESS entities that share the same area code as a protected green space
    merged_df = pd.merge(
        df_ess[['n_sq_ess', 'n_sq_ca']], 
        df_evp[['n_sq_ca']], 
        on='n_sq_ca', 
        how='inner'
    )

    # Get unique economic activity codes (n_sq_ess)
    unique_ess_codes = merged_df['n_sq_ess'].unique().tolist()

    # Output formatting
    if not unique_ess_codes:
        print("No economic activity codes in the social and solidarity economy were found to have a protected green space.")
    else:
        if len(unique_ess_codes) > 10:
            print(f"Total unique ESS codes found: {len(unique_ess_codes)}")
            print("First 5 examples:", unique_ess_codes[:5])
        else:
            print(f"Unique economic activity codes in the ESS with a protected green space: {unique_ess_codes}")

if __name__ == "__main__":
    solve()
